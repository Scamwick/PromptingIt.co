Try It Free →
app.ts
// Initialize Prompting It SDK
import { PromptingIt } from '@promptingit/sdk';

const client = new PromptingIt({
  apiKey: process.env.PROMPTINGIT_KEY
});

// Execute versioned prompt with tracking
const result = await client.execute({
  prompt: 'code-review@v2.4.1',
  model: 'claude-3-opus',
  variables: { code: userCode }
});

console.log(result.metrics);
