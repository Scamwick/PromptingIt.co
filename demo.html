integration.ts
// Install: npm install @promptingit/sdk

import { PromptingIt } from '@promptingit/sdk';

const client = new PromptingIt({
  apiKey: process.env.PROMPTINGIT_KEY
});

// Execute versioned prompt
const response = await client.execute({
  prompt: 'customer-support@v2.1',
  model: 'claude-3-sonnet',
  variables: {
    customerName: user.name,
    issueType: ticket.category
  }
});

// Metrics auto-tracked
console.log(response.metrics);
// { tokens: 523, cost: 0.008, latency: 187 }
